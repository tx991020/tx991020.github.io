#python 单例实现

class Animal(object):
    __instance = None

    def __new__(cls, *args):
        if cls.__instance is None:

            cls.__instance = object.__new__(cls)
        return cls.__instance
    def __init__(self,name):
        self.name = name


cat = Animal('miao')
print(cat)

print(cat.name)
