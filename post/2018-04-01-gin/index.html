package main

import (
	"hdutil"
	"github.com/gin-gonic/gin"
	"net/http"
	"fmt"
	"reflect"
	"logic/pkg/dao"
)

func main() {
	r := gin.Default()
	r.Use(hdutil.GJsonResponse)

	r.GET(
		`user/:customId`,
		hdutil.GPathOptionalString("customId"),
		GetData1,
	)

	r.POST(
		``,
		hdutil.GRequestBodyObject(reflect.TypeOf(dao.Hetaojun{}), "json"),
		PostData,
	)
	r.GET(
		`/homeByFilter`,
		hdutil.GQueryRequireString("filter"),
		hdutil.GQueryOptionalIntDefault("current", 1),
		hdutil.GQueryOptionalIntDefault("pageSize", 10),
		GetByFilter,
	)
	r.PUT(
		`/:pingId`,
		hdutil.GPathRequireInt("hetaojunId"),
		hdutil.GRequestBodyMap,
		UpdateData,
	)

	r.Run(":4000") // listen and serve on 0.0.0.0:8080
}

func GetData(c *gin.Context)  {

	qq := map[string]string{
		"a": "aa",
		"b": "bb",
	}

	c.Data(http.StatusOK, "", hdutil.GR(&hdutil.Error{0, ""}, qq))
}


func GetData1(c *gin.Context)  {
	u := c.MustGet("customId").(string)

	fmt.Println(111,u)
	c.Data(http.StatusOK, "", hdutil.GR(&hdutil.Error{0, ""},u))
}
